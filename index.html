    <input type="text" id="nombre" placeholder="Nombre o número">

    <input type="date" id="fecha">

    <button onclick="agregarVaca()">Guardar</button>

  </div>


  <div class="card">

    <h2>Listado</h2>

    <div id="lista"></div>

  </div>


</div>


<script>

  let vacas = JSON.parse(localStorage.getItem('vacas')) || [];


  function guardar() {

    localStorage.setItem('vacas', JSON.stringify(vacas));

  }


  function agregarVaca() {

    const nombre = document.getElementById('nombre').value;

    const fecha = document.getElementById('fecha').value;


    if (!nombre) {

      alert('Ponle nombre o número');

      return;

    }


    vacas.push({ nombre, fecha });

    guardar();

    mostrar();


    document.getElementById('nombre').value = '';

    document.getElementById('fecha').value = '';

  }


  function eliminar(index) {

    vacas.splice(index, 1);

    guardar();

    mostrar();

  }


  function mostrar() {

    const lista = document.getElementById('lista');

    lista.innerHTML = '';


    vacas.forEach((vaca, i) => {

      lista.innerHTML += `

        <div class="vaca">

          <strong>${vaca.nombre}</strong><br>

          Última fecha: ${vaca.fecha || 'Sin registro'}<br>

          <button onclick="eliminar(${i})">Eliminar</button>

        </div>

      `;

    });

  }


  mostrar();

</script>
