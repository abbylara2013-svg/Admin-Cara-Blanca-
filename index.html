import React, { useState, useEffect, useMemo } from 'react';

import { 

  Plus, 

  Search, 

  Calendar, 

  Activity, 

  Clipboard, 

  ChevronRight, 

  Milk, 

  LogOut,

  ArrowLeft,

  Sparkles,

  BrainCircuit,

  Stethoscope,

  TrendingUp,

  Zap,

  Trash2

} from 'lucide-react';

import { initializeApp } from 'firebase/app';

import { 

  getFirestore, 

  collection, 

  doc, 

  onSnapshot, 

  updateDoc, 

  deleteDoc, 

  addDoc 

} from 'firebase/firestore';

import { 

  getAuth, 

  signInAnonymously, 

  signInWithCustomToken, 

  onAuthStateChanged 

} from 'firebase/auth';


// --- CONFIGURACIÓN DE FIREBASE ---

const firebaseConfig = JSON.parse(__firebase_config);

const app = initializeApp(firebaseConfig);

const auth = getAuth(app);

const db = getFirestore(app);

const appId = typeof __app_id !== 'undefined' ? __app_id : 'api-cara-blanca-v3';

const apiKey = ""; 


export default function App() {

  const [user, setUser] = useState(null);

  const [loading, setLoading] = useState(true);

  const [view, setView] = useState('dashboard'); 

  const [cows, setCows] = useState([]);

  const [selectedCow, setSelectedCow] = useState(null);

  const [searchTerm, setSearchTerm] = useState('');

  const [aiLoading, setAiLoading] = useState(false);

  const [aiResult, setAiResult] = useState("");


  // Autenticación

  useEffect(() => {

    const initAuth = async () => {

      try {

        if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {

          await signInWithCustomToken(auth, __initial_auth_token);

        } else {

          await signInAnonymously(auth);

        }

      } catch (error) {

        console.error("Auth error:", error);

      }

    };

    initAuth();

    const unsubscribe = onAuthStateChanged(auth, (u) => {

      setUser(u);

      setLoading(false);

    });

    return () => unsubscribe();

  }, []);


  // Escucha de Datos en tiempo real

  useEffect(() => {

    if (!user) return;

    const cowsRef = collection(db, 'artifacts', appId, 'public', 'data', 'cows');

    const unsubCows = onSnapshot(cowsRef, (snapshot) => {

      const data = snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() }));

      setCows(data);

    }, (err) => console.error("Error loading cows:", err));

    return () => unsubCows();

  }, [user]);


  // --- INTEGRACIÓN GEMINI API (Reportes Directos) ---

  const callGemini = async (prompt, systemPrompt = "Eres un consultor ganadero técnico experto en medicina bovina y reproducción.") => {

    setAiLoading(true);

    let retries = 0;

    const maxRetries = 5;

    while (retries < maxRetries) {

      try {

        const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {

          method: 'POST',

          headers: { 'Content-Type': 'application/json' },

          body: JSON.stringify({

            contents: [{ parts: [{ text: prompt }] }],

            systemInstruction: { parts: [{ text: systemPrompt }] }

          })

        });

        if (!response.ok) throw new Error('API Error');

        const result = await response.json();

        const text = result.candidates?.[0]?.content?.parts?.[0]?.text;

        setAiLoading(false);

        return text;

      } catch (error) {

        retries++;

        await new Promise(resolve => setTimeout(resolve, Math.pow(2, retries) * 1000));

      }

    }

    setAiLoading(false);

    return "No se pudo generar el análisis en este momento. Inténtelo más tarde.";

  };


  const getSmartInsight = async (cow, insightType) => {

    let prompt = "";

    if (insightType === 'salud') {

      prompt = `Genera un reporte veterinario conciso para la vaca #${cow.identificador}. Estado: ${cow.estado}. Analiza riesgos sanitarios basados en su estado y sugiere 3 medidas preventivas clave.`;

    } else {

      prompt = `Realiza un análisis reproductivo para la vaca #${cow.identificador}. Estado Reproductivo: ${cow.estadoReproduccion}. Explica qué esperar en esta etapa y qué cuidados especiales requiere para optimizar la fertilidad.`;

    }

    const result = await callGemini(prompt);

    setAiResult(result);

    setView('ai_report');

  };


  // --- LÓGICA DE NEGOCIO ---

  const stats = useMemo(() => {

    return {

      activas: cows.filter(c => c.estado === 'activa').length,

      preñadas: cows.filter(c => c.estadoReproduccion === 'preñada').length,

      secas: cows.filter(c => c.estado === 'seca').length,

    };

  }, [cows]);


  const saveCow = async (cowData) => {

    if (!user) return;

    const { id, ...payload } = cowData;

    // Limpieza de campos undefined para evitar errores de Firebase

    const cleanPayload = Object.fromEntries(

      Object.entries(payload).filter(([_, v]) => v !== undefined)

    );

    

    const cowsRef = collection(db, 'artifacts', appId, 'public', 'data', 'cows');

    try {

      if (id) {

        await updateDoc(doc(cowsRef, id), cleanPayload);

      } else {

        await addDoc(cowsRef, { 

          ...cleanPayload, 

          historialTratamientos: [],

          fechaCreacion: new Date().toISOString() 

        });

      }

      setView('list');

    } catch (e) {

      console.error("Error saving cow:", e);

    }

  };


  const deleteCow = async (id) => {

    if (!user || !id) return;

    try {

      await deleteDoc(doc(db, 'artifacts', appId, 'public', 'data', 'cows', id));

      setView('list');

    } catch (e) {

      console.error("Error deleting cow:", e);

    }

  };


  // --- COMPONENTES DE VISTA ---


  const Dashboard = () => (

    <div className="p-6 pb-32 animate-in fade-in duration-500">

      <header className="mb-8">

        <p className="text-blue-600 font-bold text-xs uppercase tracking-widest">Sistema Ganadero</p>

        <h1 className="text-4xl font-black text-slate-800 tracking-tight leading-none">ApiCaraBlanca</h1>

      </header>


      <div className="grid grid-cols-2 gap-4 mb-8">

        <StatCard label="Activas" val={stats.activas} icon={<Activity size={20}/>} color="bg-blue-600 text-white" />

        <StatCard label="Gestación" val={stats.preñadas} icon={<Calendar size={20}/>} color="bg-emerald-600 text-white" />

        <StatCard label="Secas" val={stats.secas} icon={<Milk size={20}/>} color="bg-slate-800 text-white" />

        <StatCard label="Total Hato" val={cows.length} icon={<Clipboard size={20}/>} color="bg-slate-200 text-slate-600" />

      </div>


      <div className="space-y-4">

        <h2 className="font-black text-slate-800 text-lg ml-2">Acciones Rápidas</h2>

        <button 

          onClick={() => {setSelectedCow(null); setView('form');}} 

          className="w-full bg-white border-2 border-dashed border-slate-200 p-8 rounded-[2.5rem] flex flex-col items-center justify-center gap-2 text-slate-400 font-bold active:bg-slate-50 transition-all"

        >

          <div className="p-3 bg-slate-50 rounded-full text-slate-300"><Plus size={32} /></div>

          Registrar nuevo animal

        </button>

        

        <button 

          onClick={() => setView('list')} 

          className="w-full bg-blue-50 p-6 rounded-3xl flex items-center justify-between text-blue-700 font-bold active:scale-95 transition-all"

        >

          <div className="flex items-center gap-3"><Search size={20}/> Ver lista de ganado</div>

          <ChevronRight size={20} />

        </button>

      </div>

    </div>

  );


  const StatCard = ({ label, val, icon, color }) => (

    <div className={`${color} p-6 rounded-[2.5rem] shadow-xl shadow-current/5 flex flex-col gap-3`}>

      <div className="opacity-70">{icon}</div>

      <div>

        <p className="text-3xl font-black leading-none">{val}</p>

        <p className="text-[10px] font-bold uppercase tracking-widest opacity-60 mt-1">{label}</p>

      </div>

    </div>

  );


  const CowList = () => (

    <div className="p-6 pb-32 animate-in slide-in-from-right duration-300">

      <div className="flex items-center gap-3 mb-6">

        <button onClick={() => setView('dashboard')} className="p-4 bg-white rounded-2xl shadow-sm border border-slate-100"><ArrowLeft size={20}/></button>

        <div className="relative flex-1">

          <Search className="absolute left-4 top-4 text-slate-400" size={20} />

          <input 

            type="text" 

            placeholder="Buscar por ID..." 

            className="w-full bg-white py-4 pl-12 pr-4 rounded-2xl outline-none shadow-sm border border-slate-100 font-bold text-slate-700" 

            value={searchTerm} 

            onChange={(e) => setSearchTerm(e.target.value)} 

          />

        </div>

      </div>

      

      <div className="space-y-3">

        {cows.filter(c => c.identificador.includes(searchTerm)).map(cow => (

          <div 

            key={cow.id} 

            onClick={() => {setSelectedCow(cow); setView('detail');}} 

            className="bg-white p-5 rounded-[2rem] flex items-center justify-between border border-slate-50 shadow-sm active:scale-[0.98] transition-all cursor-pointer"

          >

            <div className="flex items-center gap-4">

              <div className="w-16 h-16 bg-slate-50 rounded-2xl flex items-center justify-center font-black text-blue-600 text-xl border border-slate-100">

                {cow.identificador.slice(-2)}

              </div>

              <div>

                <h3 className="font-black text-slate-800 text-lg">#{cow.identificador}</h3>

                <div className="flex gap-2 mt-0.5">

                  <span className={`text-[9px] px-2 py-0.5 rounded-full font-black uppercase tracking-widest ${cow.estado === 'activa' ? 'bg-green-100 text-green-700' : 'bg-slate-100 text-slate-500'}`}>

                    {cow.estado}

                  </span>

                  <span className="text-[9px] px-2 py-0.5 rounded-full font-black uppercase tracking-widest bg-blue-100 text-blue-700">

                    {cow.estadoReproduccion}

                  </span>

                </div>

              </div>

            </div>

            <ChevronRight className="text-slate-300" />

          </div>

        ))}

        {cows.length === 0 && (

          <div className="py-20 text-center text-slate-300 font-bold">No hay animales registrados aún.</div>

        )}

      </div>

    </div>

  );


  const CowDetail = () => (

    <div className="pb-32 animate-in slide-in-from-bottom duration-400">

      <div className="bg-slate-900 p-10 pt-16 text-white rounded-b-[4rem] relative overflow-hidden shadow-2xl">

        <button onClick={() => setView('list')} className="absolute top-8 left-6 p-3 bg-white/10 rounded-xl hover:bg-white/20 transition-colors"><ArrowLeft size={20}/></button>

        <div className="flex flex-col items-center">

          <div className="w-28 h-28 bg-white/10 rounded-[3rem] flex items-center justify-center text-5xl font-black mb-6 backdrop-blur-md border border-white/20">

            {selectedCow.identificador.slice(-2)}

          </div>

          <h2 className="text-4xl font-black tracking-tight">#{selectedCow.identificador}</h2>

          <p className="opacity-40 font-bold uppercase text-[10px] tracking-[0.3em] mt-2">{selectedCow.nombre || 'Vaca sin nombre'}</p>

        </div>

      </div>


      <div className="p-6 space-y-6">

        <div className="grid grid-cols-2 gap-4">

          <button 

            onClick={() => getSmartInsight(selectedCow, 'salud')} 

            className="bg-white p-6 rounded-[2.5rem] shadow-md border border-slate-50 flex flex-col items-center gap-3 active:bg-slate-50 transition-all group"

          >

            <div className="p-3 bg-red-50 text-red-500 rounded-2xl group-hover:scale-110 transition-transform"><Stethoscope size={28} /></div>

            <span className="font-black text-[10px] uppercase text-slate-500 tracking-wider">Reporte Salud ✨</span>

          </button>

          <button 

            onClick={() => getSmartInsight(selectedCow, 'repro')} 

            className="bg-white p-6 rounded-[2.5rem] shadow-md border border-slate-50 flex flex-col items-center gap-3 active:bg-slate-50 transition-all group"

          >

            <div className="p-3 bg-indigo-50 text-indigo-500 rounded-2xl group-hover:scale-110 transition-transform"><TrendingUp size={28} /></div>

            <span className="font-black text-[10px] uppercase text-slate-500 tracking-wider">Análisis Repro ✨</span>

          </button>

        </div>


        <div className="bg-white p-8 rounded-[3rem] shadow-sm border border-slate-100 grid grid-cols-2 gap-y-8">

            <InfoBox label="Estado Vital" val={selectedCow.estado} />

            <InfoBox label="Fertilidad" val={selectedCow.estadoReproduccion} />

            <InfoBox label="Raza Sugerida" val={selectedCow.raza || 'Holando'} />

            <InfoBox label="Registro" val={new Date().toLocaleDateString()} />

        </div>


        <div className="flex gap-4">

          <button onClick={() => setView('form')} className="flex-1 bg-slate-800 text-white py-5 rounded-3xl font-black text-sm active:scale-95 transition-all">Editar Ficha</button>

          <button onClick={() => { if(confirm('¿Eliminar registro?')) deleteCow(selectedCow.id); }} className="p-5 bg-red-50 text-red-500 rounded-3xl active:scale-95 transition-all"><Trash2 /></button>

        </div>

      </div>

    </div>

  );


  const InfoBox = ({ label, val }) => (

    <div>

      <p className="text-[10px] font-bold text-slate-300 uppercase tracking-widest mb-1">{label}</p>

      <p className="font-black text-slate-700 text-sm uppercase">{val}</p>

    </div>

  );


  const CowForm = () => {

    const [idF, setIdF] = useState(selectedCow?.identificador || '');

    const [nm, setNm] = useState(selectedCow?.nombre || '');

    const [st, setSt] = useState(selectedCow?.estado || 'activa');

    const [rp, setRp] = useState(selectedCow?.estadoReproduccion || 'vacía');


    return (

      <div className="p-8 pb-32 animate-in fade-in duration-500">

        <div className="flex items-center gap-4 mb-10">

          <button onClick={() => setView(selectedCow ? 'detail' : 'dashboard')} className="p-3 bg-slate-100 rounded-2xl"><ArrowLeft size={20}/></button>

          <h2 className="text-3xl font-black text-slate-800">{selectedCow ? 'Actualizar' : 'Registro'}</h2>

        </div>


        <div className="space-y-6">

          <div className="space-y-2">

            <label className="text-xs font-black uppercase text-slate-400 ml-4 tracking-widest">ID de Orejera</label>

            <input className="w-full p-6 rounded-[2.5rem] bg-white border-2 border-slate-100 outline-none font-black text-3xl text-blue-600 shadow-inner focus:border-blue-300 transition-all" placeholder="0000" value={idF} onChange={e => setIdF(e.target.value)} />

          </div>

          

          <div className="space-y-2">

            <label className="text-xs font-black uppercase text-slate-400 ml-4 tracking-widest">Nombre</label>

            <input className="w-full p-6 rounded-[2.5rem] bg-white border-2 border-slate-100 outline-none font-bold text-slate-700 shadow-inner" placeholder="Ej. Muñeca" value={nm} onChange={e => setNm(e.target.value)} />

          </div>


          <div className="grid grid-cols-2 gap-4">

            <div className="space-y-2">

              <label className="text-xs font-black uppercase text-slate-400 ml-4 tracking-widest">Estado</label>

              <select className="w-full p-6 rounded-[2rem] bg-white border-2 border-slate-100 outline-none font-bold appearance-none text-slate-700" value={st} onChange={e => setSt(e.target.value)}>

                <option value="activa">Activa</option>

                <option value="seca">Seca</option>

                <option value="enferma">Enferma</option>

              </select>

            </div>

            <div className="space-y-2">

              <label className="text-xs font-black uppercase text-slate-400 ml-4 tracking-widest">Reproducción</label>

              <select className="w-full p-6 rounded-[2rem] bg-white border-2 border-slate-100 outline-none font-bold appearance-none text-slate-700" value={rp} onChange={e => setRp(e.target.value)}>

                <option value="vacía">Vacía</option>

                <option value="inseminada">Inseminada</option>

                <option value="preñada">Preñada</option>

              </select>

            </div>

          </div>


          <button 

            onClick={() => saveCow({ id: selectedCow?.id, identificador: idF, nombre: nm, estado: st, estadoReproduccion: rp })} 

            className="w-full bg-blue-600 text-white py-6 rounded-[2.5rem] font-black text-xl shadow-2xl shadow-blue-100 active:scale-95 transition-all mt-8"

          >

            {selectedCow ? 'Confirmar cambios' : 'Guardar en el hato'}

          </button>

        </div>

      </div>

    );

  };


  const AIReportView = () => (

    <div className="p-6 pb-32 bg-slate-50 min-h-screen">

      <div className="flex items-center gap-4 mb-8">

        <button onClick={() => setView('detail')} className="bg-white p-4 rounded-3xl shadow-sm border border-slate-100"><ArrowLeft size={20}/></button>

        <h2 className="text-xl font-black text-slate-800">Resultado del Análisis ✨</h2>

      </div>

      

      <div className="bg-white rounded-[3.5rem] p-10 shadow-2xl shadow-slate-200 border border-slate-50 relative overflow-hidden">

        <div className="absolute top-0 right-0 p-8 opacity-5 text-indigo-600"><BrainCircuit size={120}/></div>

        

        <div className="w-16 h-16 bg-indigo-600 rounded-[1.5rem] flex items-center justify-center text-white mb-8 shadow-xl shadow-indigo-100">

          <Zap size={32} />

        </div>

        

        {aiLoading ? (

          <div className="flex flex-col items-center py-24">

            <div className="w-14 h-14 border-4 border-slate-100 border-t-indigo-600 rounded-full animate-spin mb-6"></div>

            <p className="text-slate-400 font-black uppercase tracking-widest text-[10px] animate-pulse">Consultando a Gemini AI...</p>

          </div>

        ) : (

          <div className="prose prose-slate max-w-none">

            <p className="text-slate-700 leading-relaxed font-medium text-lg whitespace-pre-wrap">

              {aiResult}

            </p>

          </div>

        )}

        

        <button 

          onClick={() => setView('detail')} 

          className="w-full bg-slate-900 text-white py-5 rounded-3xl font-black mt-10 shadow-lg active:scale-95 transition-all"

        >

          Cerrar reporte

        </button>

      </div>

    </div>

  );


  if (loading) return (

    <div className="h-screen bg-slate-50 flex flex-col items-center justify-center font-black animate-pulse text-slate-300">

      <div className="w-12 h-12 border-4 border-slate-200 border-t-blue-500 rounded-full animate-spin mb-4"></div>

      <p className="tracking-[0.2em] text-xs">SINCRONIZANDO HATO</p>

    </div>

  );


  return (

    <div className="max-w-md mx-auto min-h-screen bg-slate-50 shadow-2xl relative font-sans overflow-x-hidden">

      {view === 'dashboard' && <Dashboard />}

      {view === 'list' && <CowList />}

      {view === 'detail' && <CowDetail />}

      {view === 'form' && <CowForm />}

      {view === 'ai_report' && <AIReportView />}


      <nav className="fixed bottom-0 left-0 right-0 max-w-md mx-auto bg-white/90 backdrop-blur-xl p-5 flex justify-around border-t border-slate-100 z-50 rounded-t-[2.5rem]">

        <NavBtn active={view === 'dashboard'} icon={<Activity/>} label="Inicio" onClick={() => setView('dashboard')} />

        <NavBtn active={view === 'list'} icon={<Search/>} label="El Hato" onClick={() => setView('list')} />

        <NavBtn active={false} icon={<LogOut/>} label="Salir" onClick={() => {}} />

      </nav>

    </div>

  );

}


const NavBtn = ({ active, icon, label, onClick }) => (

  <button onClick={onClick} className={`flex flex-col items-center gap-1 transition-all ${active ? 'text-blue-600 scale-110' : 'text-slate-400'}`}>

    {icon}

    <span className="text-[9px] font-black uppercase tracking-tighter">{label}</span>

  </button>

);
