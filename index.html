<!DOCTYPE html>

<html>

<head>

  <title>Admin Cara Blanca PRO</title>


  <script type="module">

    import { initializeApp } from "https://www.gstatic.com/firebasejs/10.12.2/firebase-app.js";

    import { getFirestore, collection, addDoc, getDocs } from "https://www.gstatic.com/firebasejs/10.12.2/firebase-firestore.js";


    // 🔴 AQUÍ PEGAS TU CONFIG REAL

    const firebaseConfig = {

      apiKey: "AQUI_TU_API_KEY",

      authDomain: "AQUI",

      projectId: "AQUI",

      storageBucket: "AQUI",

      messagingSenderId: "AQUI",

      appId: "AQUI"

    };


    const app = initializeApp(firebaseConfig);

    const db = getFirestore(app);


    async function agregarVaca() {

      const nombre = document.getElementById('nombre').value;

      if (!nombre) return alert("Pon nombre");


      await addDoc(collection(db, "vacas"), {

        nombre: nombre,

        fecha: new Date().toISOString()

      });


      cargar();

    }


    async function cargar() {

      const lista = document.getElementById('lista');

      lista.innerHTML = "";


      const querySnapshot = await getDocs(collection(db, "vacas"));

      querySnapshot.forEach(doc => {

        lista.innerHTML += `<div>${doc.data().nombre}</div>`;

      });

    }


    window.agregarVaca = agregarVaca;

    window.onload = cargar;

  </script>

</head>


<body>

  <h1>🐄 Cara Blanca PRO</h1>


  <input id="nombre" placeholder="Nombre vaca">

  <button onclick="agregarVaca()">Guardar</button>


  <h2>Listado</h2>

  <div id="lista"></div>


</body>

</html>
